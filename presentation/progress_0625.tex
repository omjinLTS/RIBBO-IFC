\section{Overview}

\begin{frame}{Overview}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.50\textwidth}
            \textbf{Problem} \hfill {\scriptsize Lee et al., TCOM 2025}
            \begin{tightitemize}
                \item IFC power control, $\mathbf{x}\in[0,1]^D$, Rayleigh fading
                \item \textbf{SE} (spectral efficiency): optimum at box \emph{corners} (links full-on or off)
                \item \textbf{EE} (energy efficiency): optimum in the \emph{interior} (mid power)
            \end{tightitemize}

            \vspace{0.5em}
            \textbf{Method}
            \begin{tightitemize}
                \item RIBBO (Reinforced In-Context Black-Box Optimization): a transformer that learns \emph{how to optimize} from past attempts
                \item Key parts: regret-to-go (RTG) token, hindsight regret relabeling (HRR)
            \end{tightitemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Reference solution} {\scriptsize(defines the gap)}
            \begin{tightitemize}
                \item SE: WMMSE (classical solver; \emph{local}-optimal)
                \item EE: $D{=}3$ exhaustive grid; $D{=}10/20$ multi-start gradient (near-global)
                \item \textbf{Optimality gap (\%)} $=100\times\frac{\text{ref}-\text{achieved}}{\text{ref}}$; \textbf{lower is better} ($0=$ reference)
            \end{tightitemize}

            \vspace{0.5em}
            \textbf{Evaluation}
            \begin{tightitemize}
                \item Unseen channels: new Rayleigh $\mathrm{CN}(0,1)$ realizations (same distribution)
                \item 300 steps; behavior-algorithm (BA) pool: \textbf{3BA} $=$ Random, Hill-Climbing, CMA-ES, unless noted
            \end{tightitemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Result 4 --- Robustness Checks}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.46\textwidth}
            \begin{tightitemize}
                \item Random search ($=$ Lee et al.'s \emph{Brute-force} baseline) plateaus far from the optimum; \textbf{RIBBO reaches regions uniform sampling cannot}
                \item Smaller gap than random on \emph{every} channel at $D{=}10$ (nearly all at $D{=}20$) --- not merely on average
                \item Statistically stable (narrow 95\% confidence intervals, $n{=}200$--$500$)
                \item The EE advantage persists at $D{=}20$, on every channel
                \item A local search reaches a near-zero gap from \emph{any} start; with the model and gradients the problem is easy, so RIBBO's value is the \emph{model-free} setting
            \end{tightitemize}
        \end{column}

\begin{frame}{Result 5 --- Cross-Problem Training Does Not Transfer}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.42\textwidth}
            \textbf{Use a Rastrigin-trained model on IFC $D{=}10$ SE (no retraining)}
            \par\vspace{0.3em}
            \scriptsize
            \begin{tabular}{lr}
                \toprule
                Model                               & gap (\%) $\downarrow$ \\
                \midrule
                Rastrigin model, unchanged          & 68.2                  \\
                Rastrigin model, \textbf{rescaled}  & 68.8                  \\
                IFC model (reference)               & \textbf{55.1}         \\
                Random                              & $\sim$57              \\
                \bottomrule
            \end{tabular}

            \vspace{0.5em}
            \begin{tightitemize}
                \item Rescaling the target does \emph{not} help
                \item Our earlier hypothesis (a normalization mismatch) proved incorrect
                \item Weak SE 3BA setting here, so IFC-vs-random is small --- the point is \emph{in-domain vs.\ transfer}
                \item $\Rightarrow$ \textbf{in-domain IFC training is required}
            \end{tightitemize}
        \end{column}

\section{Insights}

\begin{frame}{Insights --- When Does RIBBO Help?}
    \vspace{0.5em}
    \begin{itemize}
        \setlength{\itemsep}{0.8em}
        \item \textbf{The effect depends on the problem and its scale.} Simple methods already do well for SE $D{=}3$; RIBBO's largest margin is for EE (48.7 pts at $D{=}10$).
        \item \textbf{Classical solvers need the channel model; RIBBO uses only scores.} WMMSE / fractional programming (FP) reach it with the model, and a local search also closed the gap --- so RIBBO's clearest role is \textbf{model-free}.
        \item \textbf{Which training algorithms help is problem-dependent.} Removing weak ones helped SE at high $D$ but hurt EE.
        \item \textbf{With a fixed setup the gap grows with $D$;} more data, more steps, and a better algorithm choice reduce it (SE).
        \item \textbf{Cross-problem transfer did not work:} in-domain IFC training was required.
    \end{itemize}
\end{frame}
